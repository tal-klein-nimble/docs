### Inspect parameter schema

Review required and optional parameters:

<CodeGroup>

```python Python
from nimble import Nimble

nimble = Nimble(api_key="YOUR-API-KEY")

template = nimble.get_template("amazon_pdp")

# Review parameter schema
print("Required parameters:")
for field_name, field_info in template["input_schema"].items():
    if field_info.get("required"):
        print(f"  - {field_name}: {field_info['type']} - {field_info['description']}")

print("\nOptional parameters:")
for field_name, field_info in template["input_schema"].items():
    if not field_info.get("required"):
        print(f"  - {field_name}: {field_info['type']} - {field_info['description']}")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

const template = await nimble.getTemplate("amazon_pdp");

// Review parameter schema
console.log("Required parameters:");
Object.entries(template.input_schema).forEach(([fieldName, fieldInfo]) => {
  if (fieldInfo.required) {
    console.log(`  - ${fieldName}: ${fieldInfo.type} - ${fieldInfo.description}`);
  }
});

console.log("\nOptional parameters:");
Object.entries(template.input_schema).forEach(([fieldName, fieldInfo]) => {
  if (!fieldInfo.required) {

});
```

### Build dynamic forms

Generate input forms based on template schema:

<CodeGroup>

```python Python
from nimble import Nimble

nimble = Nimble(api_key="YOUR-API-KEY")

def generate_form_fields(template_name):
    """Generate form fields from template input schema"""
    template = nimble.get_template(template_name)

    form_fields = []
    for field_name, field_info in template["input_schema"].items():
        form_fields.append({
            "name": field_name,
            "type": field_info["type"],
            "label": field_info["description"],
            "required": field_info.get("required", False),
            "default": field_info.get("default"),
            "options": field_info.get("enum"),
            "placeholder": field_info.get("example")
        })

    return form_fields

# Use in UI
fields = generate_form_fields("amazon_pdp")
for field in fields:
    print(f"{field['label']}: {field['name']} ({'required' if field['required'] else 'optional'})")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function generateFormFields(templateName) {
  const template = await nimble.getTemplate(templateName);

  const formFields = Object.entries(template.input_schema).map(([fieldName, fieldInfo]) => ({
    name: fieldName,
    type: fieldInfo.type,
    label: fieldInfo.description,
    required: fieldInfo.required || false,
    default: fieldInfo.default,
    options: fieldInfo.enum,
    placeholder: fieldInfo.example
  }));

  return formFields;
}

// Use in UI
const fields = await generateFormFields("amazon_pdp");

  console.log(`${field.label}: ${field.name} (${field.required ? "required" : "optional"})`);
});
```

### Validate parameters

Check parameters against template schema before execution:

<CodeGroup>

```python Python
from nimble import Nimble
import re

nimble = Nimble(api_key="YOUR-API-KEY")

def validate_params(template_name, params):
    """Validate params against template schema"""
    template = nimble.get_template(template_name)
    errors = []

    # Check required fields
    for field_name, field_info in template["input_schema"].items():
        if field_info.get("required") and field_name not in params:
            errors.append(f"Missing required field: {field_name}")

    # Validate patterns
    for field_name, value in params.items():
        field_info = template["input_schema"].get(field_name, {})
        pattern = field_info.get("pattern")
        if pattern and not re.match(pattern, value):
            errors.append(f"Invalid format for {field_name}: {value}")

    return errors

# Validate before extraction
errors = validate_params("amazon_pdp", {
    "asin": "B08N5WRWNW",
    "country": "US"
})

if errors:
    print("Validation errors:")
    for error in errors:
        print(f"  - {error}")
else:
    print("✓ All parameters valid")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function validateParams(templateName, params) {
  const template = await nimble.getTemplate(templateName);
  const errors = [];

  // Check required fields
  Object.entries(template.input_schema).forEach(([fieldName, fieldInfo]) => {
    if (fieldInfo.required && !(fieldName in params)) {
      errors.push(`Missing required field: ${fieldName}`);
    }
  });

  // Validate patterns
  Object.entries(params).forEach(([fieldName, value]) => {
    const fieldInfo = template.input_schema[fieldName] || {};
    const pattern = fieldInfo.pattern;
    if (pattern && !new RegExp(pattern).test(value)) {
      errors.push(`Invalid format for ${fieldName}: ${value}`);
    }
  });

  return errors;
}

// Validate before extraction
const errors = await validateParams("amazon_pdp", {
  asin: "B08N5WRWNW",
  country: "US"
});

if (errors.length > 0) {
  console.log("Validation errors:");
  errors.forEach(error => console.log(`  - ${error}`));
} else {
  console.log("✓ All parameters valid");
}
```

</CodeGroup>

```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function compareTemplates(templateNames) {

        outputs: Object.keys(template.output_schema).length,
```

## Best practices

### Cache template information

Reduce API calls by caching template data:

```python
import time

class TemplateCache:
    def __init__(self, nimble_client, ttl=86400):  # 24 hour default
        self.client = nimble_client
        self.ttl = ttl
        self.cache = {}
        self.last_update = {}

    def get_template(self, template_name):
        # Check cache validity
        if (template_name in self.cache and
            time.time() - self.last_update[template_name] < self.ttl):
            return self.cache[template_name]

        # Fetch and cache
        template = self.client.get_template(template_name)
        self.cache[template_name] = template
        self.last_update[template_name] = time.time()

        return template
```

### Efficient pagination

Fetch all templates efficiently:

```python
def get_all_templates(nimble_client, category=None):
    """Get all templates with automatic pagination"""
    all_templates = []
    page = 1

    while True:
        response = nimble_client.list_templates(
            category=category,
            page=page,
            limit=100
        )

        all_templates.extend(response["items"])

        # Check if there are more pages
        if page >= response["pagination"]["total_pages"]:
            break

        page += 1

    return all_templates
```
