The Templates Discovery APIs help you find and understand available templates before using them. Browse templates by category, search by keyword, and inspect detailed schemas to understand inputs and outputs.

## When to use

Use these APIs when you need to:

- **Browse available templates**: See what platforms are supported
- **Filter by category**: Find templates for specific industries
- **Search by keyword**: Locate templates for specific platforms
- **Inspect template details**: Understand input requirements and output structure
- **Validate before use**: Check supported countries and rate limits

# Two discovery APIs

## 1. List Templates

Browse and search the template catalog:

```
GET https://api.webit.live/api/v1/templates
```

## 2. Get Template Details

Get comprehensive information about a specific template:

```
GET https://api.webit.live/api/v1/templates/{template_name}
```

# List Templates API

## Parameters

| Parameter  | Type    | Description                                          | Required |
| ---------- | ------- | ---------------------------------------------------- | -------- |
| `category` | String  | Filter by category (ecommerce, social, search, etc.) | No       |
| `search`   | String  | Search templates by name or description              | No       |
| `page`     | Integer | Page number (default: 1)                             | No       |
| `limit`    | Integer | Results per page (default: 20, max: 100)             | No       |

## Usage

### List all templates

Get all available templates:

<CodeGroup>

```python Python
from nimble import Nimble

nimble = Nimble(api_key="YOUR-API-KEY")

# Get first page of templates
templates = nimble.list_templates(
    page=1,
    limit=20
)

print(f"Total templates: {templates['total']}")
for template in templates["items"]:
    print(f"- {template['name']}: {template['description']}")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

// Get first page of templates
const templates = await nimble.listTemplates({
  page: 1,
  limit: 20
});

console.log(`Total templates: ${templates.total}`);
templates.items.forEach(template => {
  console.log(`- ${template.name}: ${template.description}`);
});
```


```bash cURL
curl -X GET 'https://api.webit.live/api/v1/templates?page=1&limit=20' \
--header 'Authorization: Bearer <YOUR-API-KEY>'
```

### Inspect input schema

<CodeGroup>

```python Python
from nimble import Nimble

template = nimble.get_template("amazon_product")

print("\nOptional inputs:")
for field_name, field_info in template["input_schema"].items():
```

## Build dynamic forms

Generate input forms based on template schema:

<CodeGroup>

```python Python
from nimble import Nimble

nimble = Nimble(api_key="YOUR-API-KEY")

def generate_form_fields(template_name):
    """Generate form fields from template input schema"""
    template = nimble.get_template(template_name)

    form_fields = []
    for field_name, field_info in template["input_schema"].items():
        form_fields.append({
            "name": field_name,
            "type": field_info["type"],
            "label": field_info["description"],
            "required": field_info.get("required", False),
            "default": field_info.get("default"),
            "options": field_info.get("enum"),
            "placeholder": field_info.get("example")
        })

    return form_fields

# Use in UI
fields = generate_form_fields("amazon_product")
for field in fields:
    print(f"{field['label']}: {field['name']} ({'required' if field['required'] else 'optional'})")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function generateFormFields(templateName) {
  const template = await nimble.getTemplate(templateName);

  const formFields = Object.entries(template.input_schema).map(([fieldName, fieldInfo]) => ({
    name: fieldName,
    type: fieldInfo.type,
    label: fieldInfo.description,
    required: fieldInfo.required || false,
    default: fieldInfo.default,
    options: fieldInfo.enum,
    placeholder: fieldInfo.example
  }));

  return formFields;
}

// Use in UI

});
```

## Validate inputs

Check inputs against template schema before execution:

<CodeGroup>

```python Python
from nimble import Nimble
import re

nimble = Nimble(api_key="YOUR-API-KEY")

def validate_inputs(template_name, inputs):
    """Validate inputs against template schema"""
    template = nimble.get_template(template_name)
    errors = []

    # Check required fields
    for field_name, field_info in template["input_schema"].items():
        if field_info.get("required") and field_name not in inputs:
            errors.append(f"Missing required field: {field_name}")

    # Validate patterns
    for field_name, value in inputs.items():
        field_info = template["input_schema"].get(field_name, {})
        pattern = field_info.get("pattern")
        if pattern and not re.match(pattern, value):
            errors.append(f"Invalid format for {field_name}: {value}")

    return errors

# Validate before extraction
errors = validate_inputs("amazon_product", {
    "asin": "B08N5WRWNW",
    "country": "US"
})

if errors:
    print("Validation errors:")
    for error in errors:
        print(f"  - {error}")
else:
    print("✓ All inputs valid")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function validateInputs(templateName, inputs) {
  const template = await nimble.getTemplate(templateName);
  const errors = [];

  // Check required fields
  Object.entries(template.input_schema).forEach(([fieldName, fieldInfo]) => {
    if (fieldInfo.required && !(fieldName in inputs)) {
      errors.push(`Missing required field: ${fieldName}`);
    }
  });

  // Validate patterns
  Object.entries(inputs).forEach(([fieldName, value]) => {
    const fieldInfo = template.input_schema[fieldName] || {};
    const pattern = fieldInfo.pattern;
    if (pattern && !new RegExp(pattern).test(value)) {
      errors.push(`Invalid format for ${fieldName}: ${value}`);
    }
  });

  return errors;
}

// Validate before extraction
const errors = await validateInputs("amazon_product", {
  asin: "B08N5WRWNW",
  country: "US"
});

if (errors.length > 0) {
  console.log("Validation errors:");
  errors.forEach(error => console.log(`  - ${error}`));
} else {
  console.log("✓ All inputs valid");
}
```

</CodeGroup>

## Compare templates

Evaluate multiple templates for your use case:

<CodeGroup>

```python Python
from nimble import Nimble

nimble = Nimble(api_key="YOUR-API-KEY")

def compare_templates(template_names):
    """Compare capabilities of multiple templates"""
    comparison = []

    for name in template_names:
        template = nimble.get_template(name)
        comparison.append({
            "name": name,
            "description": template["description"],
            "inputs": len(template["input_schema"]),
            "outputs": len(template["output_schema"]),
            "countries": len(template["supported_countries"]),
            "rate_limit": template["rate_limit"]
        })

    return comparison

# Compare e-commerce templates
templates = compare_templates([
    "amazon_product",
    "walmart_product",
    "ebay_product"
])

for t in templates:
    print(f"{t['name']}: {t['inputs']} inputs, {t['outputs']} outputs, {t['countries']} countries")
```


```node Node.js
import { Nimble } from "@nimbleway/sdk";

const nimble = new Nimble({ apiKey: "YOUR-API-KEY" });

async function compareTemplates(templateNames) {
  const comparison = await Promise.all(
    templateNames.map(async (name) => {
      const template = await nimble.getTemplate(name);
      return {
        name,
        description: template.description,
        inputs: Object.keys(template.input_schema).length,
        outputs: Object.keys(template.output_schema).length,
        countries: template.supported_countries.length,
        rate_limit: template.rate_limit
      };
    })
  );

  return comparison;
}

// Compare e-commerce templates
const templates = await compareTemplates([
  "amazon_product",
  "walmart_product",
  "ebay_product"
]);

templates.forEach(t => {
  console.log(`${t.name}: ${t.inputs} inputs, ${t.outputs} outputs, ${t.countries} countries`);
});
```

# Best practices

## Cache template information

Reduce API calls by caching template data:

```python
import time

class TemplateCache:
    def __init__(self, nimble_client, ttl=86400):  # 24 hour default
        self.client = nimble_client
        self.ttl = ttl
        self.cache = {}
        self.last_update = {}

    def get_template(self, template_name):
        # Check cache validity
        if (template_name in self.cache and
            time.time() - self.last_update[template_name] < self.ttl):
            return self.cache[template_name]

        # Fetch and cache
        template = self.client.get_template(template_name)
        self.cache[template_name] = template
        self.last_update[template_name] = time.time()

        return template
```

## Efficient pagination

Fetch all templates efficiently:

```python
def get_all_templates(nimble_client, category=None):
    """Get all templates with automatic pagination"""
    all_templates = []
    page = 1

    while True:
        response = nimble_client.list_templates(
            category=category,
            page=page,
            limit=100
        )

        all_templates.extend(response["items"])

        # Check if there are more pages
        if page >= response["pagination"]["total_pages"]:
            break

        page += 1

    return all_templates
```
